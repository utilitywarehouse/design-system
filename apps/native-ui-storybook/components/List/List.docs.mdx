# List

The List component serves as the primary wrapper for List Items and List Heading components.

List item is the main building block of list. Each item has its own set of properties that you can customise and adapt to suit your needs.

## Usage

> This component should be wrapped in a `NativeUIProvider` component to ensure that the correct theme is applied.

<Canvas>
  <NativeUIProvider>
    <List headingText="Heading" headingSupportingText="Supporting text" divider>
      <ListItem text="List Item 1" onPress={() => console.log('item pressed')} />
      <ListItem text="List Item 2" onPress={() => console.log('item pressed')} />
      <ListItem text="List Item 3" onPress={() => console.log('item pressed')} />
    </List>
  </NativeUIProvider>
</Canvas>

```tsx
import { List, ListItem } from '@utilitywarehouse/native-ui';

const MyComponent = () => (
  <List headingText="Heading" headingSupportingText="Supporting text" divider>
    <ListItem text="List Item 1" onPress={() => console.log('item pressed')} />
    <ListItem text="List Item 2" onPress={() => console.log('item pressed')} />
    <ListItem text="List Item 3" onPress={() => console.log('item pressed')} />
  </List>
);
```

## Props

### `List`

| Name                  | Type                | Default  | Description                                                |
| --------------------- | ------------------- | -------- | ---------------------------------------------------------- |
| container             | `'full' \| 'card' ` | `'full'` | The container style of the list.                           |
| headingText           | string              |          | The text to display in the heading of the list.            |
| headingSupportingText | string              |          | The supporting text to display in the heading of the list. |
| divider               | boolean             | false    | Whether to display a divider below the list items.         |
| loading               | boolean             | false    | Whether to show the list items in loading state.           |
| disabled              | boolean             | false    | Whether to disable the list.                               |

### `ListHeading`

| Name           | Type   | Default | Description                                                |
| -------------- | ------ | ------- | ---------------------------------------------------------- |
| text           | string |         | The text to display in the heading of the list.            |
| supportingText | string |         | The supporting text to display in the heading of the list. |

### `ListItem`

| Name            | Type       | Default | Description                                                     |
| --------------- | ---------- | ------- | --------------------------------------------------------------- |
| text            | string     |         | The text to display in the list item.                           |
| onPress         | () => void |         | A callback function to be called when the list item is pressed. |
| supportingText  | string     |         | The supporting text to display in the list item.                |
| leadingContent  | ReactNode  |         | The leading content to display in the list item.                |
| trailingContent | ReactNode  |         | The trailing content to display in the list item.               |
| disabled        | boolean    | false   | Whether to disable the list item.                               |
| loading         | boolean    | false   | Whether to show the list item in loading state.                 |
| divider         | boolean    | false   | Whether to display a divider below the list item.               |

### Checbox List

You can also use the list to display a list of checkboxes. You can achieve this by using the `Checkbox` component as the `trailingContent` of the `ListItem`.

<Canvas>
  <NativeUIProvider>
    <CheckboxList />
  </NativeUIProvider>
</Canvas>

```tsx
import {
  List,
  ListItem,
  ListItemIcon,
  Checkbox,
  CheckboxGroup,
  CheckboxIndicator,
} from '@utilitywarehouse/native-ui';
import { UserMediumIcon } from '@utilitywarehouse/react-native-icons';

const CheckboxList = () => {
  const [values, setValues] = React.useState<Array<string>>([]);

  const handlePress = (item: string) => {
    if (values.includes(item)) {
      setValues(values.filter(value => value !== item));
    } else {
      setValues([...values, item]);
    }
  };
  return (
    <CheckboxGroup aria-label="Checkbox Group" nativeID="checkbox-group" value={values}>
      <List>
        <ListItem
          text="Joe Blogs"
          supportingText="12 Somewhere Street"
          leadingContent={<ListItemIcon as={UserMediumIcon} />}
          trailingContent={
            <Checkbox value="option1" aria-label="Label 1">
              <CheckboxIndicator />
            </Checkbox>
          }
          onPress={() => handlePress('option1')}
        />
        <ListItem
          text="Mary Poppins"
          supportingText="45 Somewhere Street"
          leadingContent={<ListItemIcon as={UserMediumIcon} />}
          trailingContent={
            <Checkbox value="option2" aria-label="Label 2">
              <CheckboxIndicator />
            </Checkbox>
          }
          onPress={() => handlePress('option2')}
        />
        <ListItem
          text="Charlie Brown"
          supportingText="78 Somewhere Street"
          leadingContent={<ListItemIcon as={UserMediumIcon} />}
          trailingContent={
            <Checkbox value="option3" aria-label="Label 3">
              <CheckboxIndicator />
            </Checkbox>
          }
          onPress={() => handlePress('option3')}
        />
        <ListItem
          text="Jane Doe"
          supportingText="90 Somewhere Street"
          leadingContent={<ListItemIcon as={UserMediumIcon} />}
          trailingContent={
            <Checkbox value="option4" aria-label="Label 4">
              <CheckboxIndicator />
            </Checkbox>
          }
          onPress={() => handlePress('option4')}
        />
      </List>
    </CheckboxGroup>
  );
};
```

For more information on the `Checbox` component, see the [`Checkbox` component documentation](/docs/native-ui-components-checkbox--docs).

### Radio List

You can also use the list to display a list of radio buttons. You can achieve this by using the `Checkbox` component as the `trailingContent` of the `ListItem`.

<Canvas>
  <NativeUIProvider>
    <RadioList />
  </NativeUIProvider>
</Canvas>

```tsx
import {
  List,
  ListItem,
  Radio,
  RadioGroup,
  RadioIndicator,
  Image,
  Box,
  useColorMode,
} from '@utilitywarehouse/native-ui';

const RadioList = () => {
  const [value, setValue] = React.useState<string>();
  const colorMode = useColorMode();

  const handlePress = (item: string) => {
    setValue(item);
  };
  return (
    <RadioGroup aria-label="Radio Group" nativeID="Radio-group" value={value}>
      <List container="card">
        <Box
          borderRadius="$xl"
          backgroundColor="$white"
          $dark-backgroundColor="$darkGrey175"
          overflow="hidden"
        >
          <ListItem
            text="**** 5016"
            supportingText="Expires 12/24"
            divider
            leadingContent={
              <Image
                source={{ uri: `/images/paymentMethods/visa/card${capitalize(colorMode)}.svg` }}

                height={42}
              />
            }
            trailingContent={
              <Radio value="option1" aria-label="Label 1">
                <RadioIndicator />
              </Radio>
            }
            onPress={() => handlePress('option1')}
          />
          <ListItem
```
